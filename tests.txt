 * - PHP 7.4+
 * - ZipArchive enabled for folder download

const APP_PASSWORD = 'ChangeThisNow123!'; // change this
const STORAGE_DIR = __DIR__ . '/storage';
const MAX_FILE_SIZE = 50 * 1024 * 1024; // 50 MB per file

session_start();

if (!is_dir(STORAGE_DIR)) {
    mkdir(STORAGE_DIR, 0755, true);
}

// =========================
// HELPERS
// =========================
function h(string $value): string {
    return htmlspecialchars($value, ENT_QUOTES, 'UTF-8');
}

function isLoggedIn(): bool {
    return !empty($_SESSION['logged_in']);
}

function requireLogin(): void {
    if (!isLoggedIn()) {
        header('Location: ' . strtok($_SERVER['REQUEST_URI'], '?'));
        exit;
    }
}

function normalizeRelativePath(string $path): string {
    $path = str_replace(['\\', "\0"], ['/', ''], $path);
    $path = trim($path, '/');
    return $path;
}

function realPathInsideStorage(string $relativePath): ?string {
    $relativePath = normalizeRelativePath($relativePath);
    $base = realpath(STORAGE_DIR);
    if ($base === false) {
        return null;
    }

    $target = $base . ($relativePath !== '' ? DIRECTORY_SEPARATOR . $relativePath : '');
    $real = realpath($target);

    if ($real === false) {
        return null;
    }

    if ($real === $base || strpos($real, $base . DIRECTORY_SEPARATOR) === 0) {
        return $real;
    }

    return null;
}

function relativeFromStorage(string $fullPath): string {
    $base = rtrim(str_replace('\\', '/', realpath(STORAGE_DIR)), '/');
    $full = str_replace('\\', '/', $fullPath);
    return ltrim(substr($full, strlen($base)), '/');
}

function sendFileDownload(string $fullPath, ?string $downloadName = null): void {
    if (!is_file($fullPath) || !is_readable($fullPath)) {
        http_response_code(404);
        exit('File not found.');
    }

    $downloadName ??= basename($fullPath);

    header('Content-Description: File Transfer');
    header('Content-Type: application/octet-stream');
    header('Content-Disposition: attachment; filename="' . rawurlencode($downloadName) . '"');
    header('Content-Length: ' . filesize($fullPath));
    header('Cache-Control: no-store, no-cache, must-revalidate');
    header('Pragma: no-cache');
    header('Expires: 0');

    while (ob_get_level()) {
        ob_end_clean();
    }

    $fp = fopen($fullPath, 'rb');
    if ($fp === false) {
        http_response_code(500);
        exit('Unable to open file.');
    }

    fpassthru($fp);
    fclose($fp);
    exit;
}

function zipFolder(string $folderPath): ?string {
    if (!class_exists('ZipArchive')) {
        return null;
    }

    $zipFile = tempnam(sys_get_temp_dir(), 'zip_');
    if ($zipFile === false) {
        return null;
    }

    // tempnam creates a file; ZipArchive needs a path, so use it directly
    $zip = new ZipArchive();
    if ($zip->open($zipFile, ZipArchive::OVERWRITE) !== true) {
        @unlink($zipFile);
        return null;
    }

    $folderPath = realpath($folderPath);
    if ($folderPath === false) {
        $zip->close();
        @unlink($zipFile);
        return null;
    }

    $folderName = basename($folderPath);

    $iterator = new RecursiveIteratorIterator(
        new RecursiveDirectoryIterator($folderPath, FilesystemIterator::SKIP_DOTS),
        RecursiveIteratorIterator::SELF_FIRST
    );

    foreach ($iterator as $item) {
        $itemPath = $item->getRealPath();
        if ($itemPath === false) {
            continue;
        }

        $localName = $folderName . '/' . ltrim(str_replace('\\', '/', substr($itemPath, strlen($folderPath))), '/');

        if ($item->isDir()) {
            $zip->addEmptyDir($localName);
        } else {
            $zip->addFile($itemPath, $localName);
        }
    }

    $zip->close();
    return $zipFile;
}

function listStorageItems(string $dir): array {
    $items = [];
    $scan = scandir($dir);
    if ($scan === false) {
        return $items;
    }

    foreach ($scan as $name) {
        if ($name === '.' || $name === '..') {
            continue;
        }

        $full = $dir . DIRECTORY_SEPARATOR . $name;
        $items[] = [
            'name' => $name,
            'path' => $full,
            'is_dir' => is_dir($full),
            'size' => is_file($full) ? filesize($full) : 0,
            'mtime' => filemtime($full) ?: 0,
        ];
    }

    usort($items, function ($a, $b) {
        if ($a['is_dir'] !== $b['is_dir']) {
            return $a['is_dir'] ? -1 : 1;
        }
        return strcasecmp($a['name'], $b['name']);
    });

    return $items;
}

function formatBytes(int $bytes): string {
    $units = ['B', 'KB', 'MB', 'GB', 'TB'];
    $i = 0;
    $size = (float)$bytes;

    while ($size >= 1024 && $i < count($units) - 1) {
        $size /= 1024;
        $i++;
    }

    return round($size, 2) . ' ' . $units[$i];
}

if (isset($_POST['login_password'])) {
    $password = (string)($_POST['login_password'] ?? '');
    if (hash_equals(APP_PASSWORD, $password)) {
        $_SESSION['logged_in'] = true;
        header('Location: ' . strtok($_SERVER['REQUEST_URI'], '?'));
        exit;
    } else {
        $loginError = 'Invalid password.';
    }

    session_destroy();
    header('Location: ' . strtok($_SERVER['REQUEST_URI'], '?'));
    exit;
}

// =========================
// DOWNLOAD ACTIONS
// =========================
if (isLoggedIn() && isset($_GET['download'])) {
    $relative = (string)$_GET['download'];
    $fullPath = realPathInsideStorage($relative);

    if ($fullPath === null || !is_file($fullPath)) {
        http_response_code(404);
        exit('File not found.');
    }

    sendFileDownload($fullPath, basename($fullPath));
}

if (isLoggedIn() && isset($_GET['download_folder'])) {
    $relative = (string)$_GET['download_folder'];
    $fullPath = realPathInsideStorage($relative);

    if ($fullPath === null || !is_dir($fullPath)) {
        http_response_code(404);
        exit('Folder not found.');
    }

    $zipPath = zipFolder($fullPath);
    if ($zipPath === null) {
        http_response_code(500);
        exit('ZIP creation failed. Make sure ZipArchive is enabled.');
    }

    $downloadName = basename($fullPath) . '.zip';
    sendFileDownload($zipPath, $downloadName);
}

// =========================
// UPLOAD ACTION
// =========================
$messages = [];
$errors = [];

if (isLoggedIn() && $_SERVER['REQUEST_METHOD'] === 'POST' && isset($_FILES['files'])) {
    requireLogin();

    if (!isset($_FILES['files']['name']) || !is_array($_FILES['files']['name'])) {
        $errors[] = 'No files received.';
    } else {
        $count = count($_FILES['files']['name']);

        for ($i = 0; $i < $count; $i++) {
            $originalName = (string)$_FILES['files']['name'][$i];
            $tmpName = (string)$_FILES['files']['tmp_name'][$i];
            $errorCode = (int)$_FILES['files']['error'][$i];
            $size = (int)$_FILES['files']['size'][$i];

            if ($errorCode === UPLOAD_ERR_NO_FILE) {
                continue;
            }

            if ($errorCode !== UPLOAD_ERR_OK) {
                $errors[] = "Upload failed for {$originalName}.";
                continue;
            }

            if ($size <= 0 || $size > MAX_FILE_SIZE) {
                $errors[] = "Invalid size for {$originalName}.";
                continue;
            }

            $safeName = basename(str_replace(["\0", '/', '\\'], '', $originalName));
            if ($safeName === '') {
                $errors[] = 'Invalid file name.';
                continue;
            }

            $destination = STORAGE_DIR . DIRECTORY_SEPARATOR . $safeName;

            if (!is_uploaded_file($tmpName)) {
                $errors[] = "Invalid upload for {$safeName}.";
                continue;
            }

            if (!move_uploaded_file($tmpName, $destination)) {
                $errors[] = "Could not save {$safeName}.";
                continue;
            }

            $messages[] = "Uploaded: {$safeName}";
        }
    }
}
